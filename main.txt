package main

import "fmt"

func cleanUpHubs(hubs *map[string]int) {
	targets := []string{}
	for channel, hub := range *hubs {
		if hub == 0 {
			targets = append(targets, channel)
		}
	}

	for _, target := range targets {
		delete(*hubs, target)
	}
}

func main() {

	hubs := map[string]int{
		"1": 1, "2": 1, "3": 0, "4": 1, "5": 0, "6": 1,
	}
	cleanUpHubs(&hubs)
	fmt.Println(hubs)
}
